#include<stdio.h>
int missingelement(int arr[],int n)
{
    int left=0,right=n - 1;
    int mid=left + right;
    while(left<=right)
    {
        if(arr[mid]!= mid + 1 && (mid==0 || arr[mid - 1]==mid))
        {
            return mid +1;
        }
        else if(arr[mid]==mid + 1)
        {
            right = mid - 1;
        }
        else{
            left = mid + 1;
        }
    }
}
int main()
{
    int arr[]={1,2,3,4,5,7};
    int n=sizeof(arr)/sizeof(arr[0]);
    int miss=missingelement(arr,n);
    printf("%d ",miss);
}
