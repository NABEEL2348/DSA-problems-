#include<stdio.h>
#include<stdlib.h>
struct node{
    int num;
    struct node *next,*prev;
};
int create(struct node **head,int data)
{
    struct node* newnode=(struct node*)malloc(sizeof(struct node));
    newnode->num=data;
    newnode->next=NULL;
    newnode->prev=NULL;
    if(*head==NULL)
    {
        *head=newnode;
    }
    else
    {
        struct node *tptr=*head;
        while(tptr->next!=NULL)
        {
            tptr=tptr->next;
        }
        tptr->next=newnode;
    }
}
int display(struct node *head)
{
    struct node *tptr=head;
    while(tptr!=NULL)
    {
        printf("%d ",tptr->num);
        tptr=tptr->next;
    }
}
int main()
{
    struct node *head=NULL;
    create(&head,40);
    create(&head,50);
    create(&head,60);
    display(head);
}
