#include <stdio.h>
#include<string.h>
int myatoi(char *str,int n)
{
    int i=0;
    int myai=1;
    int res=0;
    int sign=0;
    while(str[i]==' ')
    {
        i++;
    }
    if(str[i]=='-')
    {
        sign=-1;
        i++;
    }
    else if(str[i]=='+')
    {
        i++;
    }
    while(i!=n && str[i]!=' ')
    {
        res=(res*10) + (str[i]-'0');
        i++;
    }
    return res*sign;
}
int main()
{
    char str[]="-12345";
    int n=strlen(str);
    printf("%d",myatoi(str,n));
