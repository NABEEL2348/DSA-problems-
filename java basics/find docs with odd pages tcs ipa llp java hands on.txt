import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {
    public static void main(String args[] ) throws Exception {
        Scanner sc=new Scanner(System.in);
        int n=4;
        Book[] bk=new Book[n];
        for(int i=0;i<n;i++)
        {
            int id=sc.nextInt();
            sc.nextLine();
            String title=sc.nextLine();
            String author=sc.nextLine();
            Double price=sc.nextDouble();
            bk[i]=new Book(id,title,author,price);
        }
        Arrays.sort(bk,Comparator.comparingDouble(b->b.price));
        sortbyprice(bk,n);
    }
    static void sortbyprice(Book[] bk,int n)
    {
        double[] arr= new double[n];
        int count=0;
        for(Book book:bk)
        {
            if(count==0)
            {
                arr[count++]=book.price;
                System.out.println(book.id+" "+book.title+" "+book.author+" "+book.price);
            }
            else if(findprice(book,arr,count)==1)
            {
                System.out.println(book.id+" "+book.title+" "+book.author+" "+book.price);
                count++;
            }
        }
    }
    static int findprice(Book bk,double[] arr,int count)
    {
        for(int i=0;i<count;i++)
        {
            if(arr[i]==bk.price)
            {
                return 0;
            }
        }
        return 1;
    }
}
class Book
{
    int id;
    String title;
    String author;
    Double price;
    Book(int id,String title,String author,Double price)
    {
        this.id=id;
        this.title=title;
        this.author=author;
        this.price=price;
    }
            
}
